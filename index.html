const toggleButton = document.getElementById('theme-toggle'); const themeIcon = document.getElementById('theme-icon'); const body = document.body; // Assurez que le thème clair est initialisé au chargement de la page body.setAttribute('data-theme', 'light'); themeIcon.src = "https://cdn-icons-png.flaticon.com/512/6222/6222248.png"; themeIcon.alt = "Mode Jour"; toggleButton.addEventListener('click', () => { if (body.getAttribute('data-theme') === 'light') { body.setAttribute('data-theme', 'dark'); themeIcon.src = "https://cdn.iconscout.com/icon/free/png-256/free-moon-icon-download-in-svg-png-gif-file-formats--night-weather-climate-forecast-travels-pack-holidays-icons-1289406.png?f=webp&w=256"; themeIcon.alt = "Mode Nuit"; } else { body.setAttribute('data-theme', 'light'); themeIcon.src = "https://cdn-icons-png.flaticon.com/512/6222/6222248.png"; themeIcon.alt = "Mode Jour"; } });
